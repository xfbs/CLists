33  * => slist_t

125 /* the main slist struct. it stores a pointer to the first
126  * and the last nodes in the list, as well as the size of
127  * the list (since obtaining that is computationally expensive
128  * linear to the size of the list, as all nodes need to be
129  * traversed)
130  *
131  * FIXME: length and size may be confusing. (one is length of
132  * the list, the other is the size of the arbitrary data in
133  * each node).
134  */
135 struct slist
136 {
137 #ifdef SLIST_THREADSAFE
138  pthread_rwlock_t lock;
139 #endif
140 
141  // head of the list (first node)
142  struct array_list_node *head;
143 
144  // tail of the list (last node, for efficiency)
145  struct array_list_node *tail;
146 
147  // length of the list (how many nodes, for for efficency)
148  size_t length;
149 
150  // size of data in each node (same for all nodes)
151  size_t size;
152 };
