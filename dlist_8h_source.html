40 /* macro foreach loop implementations for dlist
41  *
42  * these macros can be used as foreach loops, with one going from
43  * beginning to end and one for traversing the list backwards (which
44  * is efficient because it's a doubly linked list, each node has a
45  * pointer to the previous node as well)
46  * assuming that list is of type dlist_t*, use them like this:
47  *
48  * void *item;
49  * dlist_foreach(list, item) {
50  * printf("%p", item);
51  * }
52  *
53  * for backwards traversal:
54  *
55  * dlist_forach_backwards(list, item) {
56  * printf("%p", item);
57  * }
58  */
59 #define dlist_foreach(list, __data) \
60  for(void *__node = list->head, *__data = ((list->head) ? \
61  list->head->data : \
62  NULL); \
63  __node != NULL; \
64  __data = ((__node = (void*)((dlist_node_t*)__node)->next) ? \
65  ((dlist_node_t*)__node)->data : \
66  NULL))
67 #define dlist_foreach_backwards(list, __data) \
68  for(void *__node = list->tail, *__data = ((list->tail) ? \
69  list->tail->data : \
70  NULL); \
71  __node != NULL; \
72  __data = ((__node = (void*)((dlist_node_t*)__node)->prev) ? \
73  ((dlist_node_t*)__node)->data : \
74  NULL))
